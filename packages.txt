def perform_calculation(matrix, calc_type, show_steps, show_viz, show_ai_tutor):
    """执行计算"""
    result = {
        'matrix': matrix,
        'type': calc_type,
        'timestamp': pd.Timestamp.now(),
        'steps': []
    }
    
    try:
        if "行列式" in calc_type:
            det_result = compute_determinant_lu(matrix)
            result['determinant'] = det_result
            
            if show_ai_tutor:
                for i, step in enumerate(det_result.get('steps', [])):
                    generate_step_explanation('lu_factorization', {
                        'k': step.get('step', i),
                        'pivot': step.get('pivot', 0)
                    })
            
        elif "逆矩阵" in calc_type:
            inv_result = compute_inverse_gauss_jordan(matrix)
            result['inverse'] = inv_result
            
        elif "特征值" in calc_type:
            eigen_result = compute_eigenvalue_qr(matrix)
            result['eigenvalues'] = eigen_result
            
        elif "线性方程组" in calc_type:
            # 生成随机b向量用于演示
            b = np.random.randint(-10, 11, matrix.shape[0])
            solve_result = solve_linear_system(matrix, b)
            result['solution'] = solve_result
            
        elif "矩阵秩" in calc_type:
            rank_result = compute_rank(matrix)
            result['rank'] = rank_result
            
        elif "全部运算" in calc_type:
            # 批量计算所有运算
            result['determinant'] = compute_determinant_lu(matrix)
            result['rank'] = compute_rank(matrix)
            try:
                result['inverse'] = compute_inverse_gauss_jordan(matrix)
            except:
                result['inverse'] = None
            try:
                result['eigenvalues'] = compute_eigenvalue_qr(matrix)
            except:
                result['eigenvalues'] = None
        
        return result
        
    except Exception as e:
        st.error(f"❌ 计算错误: {str(e)}")
        return None

def render_matrix_editor(matrix):
    """渲染矩阵编辑器"""
    if matrix is None:
        st.info("💡 请在侧边栏输入矩阵")
        return
    
    st.subheader("📊 当前矩阵")
    
    # 矩阵显示
    df = pd.DataFrame(matrix)
    st.dataframe(df, use_container_width=True, height=min(400, len(matrix) * 50))
    
    # 矩阵属性
    col1, col2, col3, col4 = st.columns(4)
    with col1:
        st.metric("维度", f"{matrix.shape[0]}×{matrix.shape[1]}")
    with col2:
        try:
            cond = np.linalg.cond(matrix)
            st.metric("条件数", f"{cond:.2e}")
        except:
            st.metric("条件数", "N/A")
    with col3:
        try:
            rank = np.linalg.matrix_rank(matrix)
            st.metric("秩", rank)
        except:
            st.metric("秩", "N/A")
    with col4:
        is_square = matrix.shape[0] == matrix.shape[1]
        st.metric("方阵", "✓" if is_square else "✗")
    
    # 矩阵操作
    st.markdown("#### 🛠️ 快速操作")
    col1, col2, col3, col4 = st.columns(4)
    with col1:
        if st.button("🔄 转置"):
            st.session_state.random_matrix = matrix.T
            st.rerun()
    with col2:
        if st.button("➕ 加单位矩阵"):
            if matrix.shape[0] == matrix.shape[1]:
                st.session_state.random_matrix = matrix + np.eye(matrix.shape[0])
                st.rerun()
    with col3:
        if st.button("✖️ 乘2"):
            st.session_state.random_matrix = matrix * 2
            st.rerun()
    with col4:
        if st.button("🎲 随机扰动"):
            noise = np.random.randn(*matrix.shape) * 0.1
            st.session_state.random_matrix = matrix + noise
            st.rerun()

def render_calculation_process(result):
    """渲染计算过程"""
    st.subheader("🔢 详细计算过程")
    
    if 'determinant' in result:
        det = result['determinant']
        st.markdown(f"### 📊 行列式计算结果")
        st.latex(f"\\det(A) = {det.get('value', 'N/A')}")
        
        if 'steps' in det:
            st.markdown("#### 📝 LU分解步骤")
            for i, step in enumerate(det['steps'][:10]):  # 最多显示10步
                with st.expander(f"步骤 {i+1}: {step.get('description', '计算')}", expanded=i==0):
                    if 'matrix' in step:
                        st.dataframe(pd.DataFrame(step['matrix']), use_container_width=True)
                    if 'formula' in step:
                        st.latex(step['formula'])
    
    if 'inverse' in result and result['inverse']:
        inv = result['inverse']
        st.markdown(f"### 🔄 逆矩阵计算结果")
        st.latex("A^{-1} = ")
        st.dataframe(pd.DataFrame(inv.get('matrix', [])), use_container_width=True)
    
    if 'eigenvalues' in result and result['eigenvalues']:
        eigen = result['eigenvalues']
        st.markdown(f"### ⚡ 特征值计算结果")
        values = eigen.get('values', [])
        for i, val in enumerate(values):
            st.latex(f"\\lambda_{{{i+1}}} = {val:.6f}")
    
    if 'solution' in result:
        sol = result['solution']
        st.markdown(f"### 📐 线性方程组求解结果")
        st.latex("x = ")
        st.dataframe(pd.DataFrame(sol.get('x', [])), use_container_width=True)

def render_visualization(result):
    """渲染可视化"""
    st.subheader("📊 可视化展示")
    
    if 'determinant' in result:
        det = result['determinant']
        if 'intermediate_states' in det:
            fig = plot_matrix_heatmap(det['intermediate_states'])
            st.plotly_chart(fig, use_container_width=True)
    
    if 'eigenvalues' in result and result['eigenvalues']:
        eigen = result['eigenvalues']
        if result['matrix'].shape == (2, 2):
            fig = plot_eigenvalue_geometry(
                eigen.get('values', []),
                eigen.get('vectors', [])
            )
            st.plotly_chart(fig, use_container_width=True)
        
        # 收敛曲线
        if 'convergence' in eigen:
            fig = plot_convergence_curve(eigen['convergence'])
            st.plotly_chart(fig, use_container_width=True)
